pipeline {
    agent any

    stages {
        stage('Checkout') {
            steps {
                git 'https://github.com/ranipulgam/webapp.git' // replace with your GitHub repo URL
            }
        }

        stage('Install Dependencies') {
            steps {
                sh 'npm install'
            }
        }

        stage('Build') {
            steps {
                echo 'Build complete'
            }
        }

        stage('Test') {
            steps {
                sh 'npm test'
            }
        }

        stage('Deploy') {
            steps {
                sh 'npm start &'
                echo 'Application started on http://localhost:3000'
            }
        }
    }

    post {
        success {
            echo 'Build and deploy succeeded!'
        }
        failure {
            echo 'Build or deploy failed.'
        }
    }
}
